ðŸ‘‰SOLUTION-1: Work in MYSQL Server
SELECT 
    CEIL(
        AVG(SALARY) - 
        AVG(REPLACE(SALARY, '0', ''))
    ) AS salary_error
FROM EMPLOYEES;

__________________________________________________________
ðŸ‘‰SOLUTION-2: Work in MSSQL SERVER or any other SQL server 
SELECT 
    CEILING(
        AVG(CAST(SALARY AS DECIMAL(18,2))) -
        AVG(
            CAST(
                NULLIF(REPLACE(CAST(SALARY AS VARCHAR), '0', ''), '') 
                AS DECIMAL(18,2)
            )
        )
    ) AS salary_error
FROM EMPLOYEES;
